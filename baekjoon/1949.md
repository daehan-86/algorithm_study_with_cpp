## 풀이
이 문제는 트리에서의 dp문제이다.  
이전에 푼거와 같이 트리의 노드를 순회하며 근접한 노드의 조건을 생각하며 풀어야 한다.  
이 문제 또한 dp 에서 메모이제이션을 어떻게 구현할지가 관건이였다.  

## 코드
```cpp
#include <iostream>
#include<vector>
#define INF 2147483647
using namespace std;
int n,a,b,cache[10001][2];
vector<int> cost(1),v[10001];
bool visit[10001];
int dp(int x,int p){
	int&ret = cache[x][p];
	if(ret) return ret;
	ret =0;
	visit[x]=1;
	for(int o : v[x]){
		if(!visit[o]){
			if(p){
				ret += dp(o,0);
			}
			else{
				ret += max(dp(o,1)+cost[o],dp(o,0));
			}
		}
	}
	visit[x]=0;
	return ret;
}
int main(void) {
	cin.tie(0);
	ios::sync_with_stdio(0);
	cin>>n;
	for(int i=0;i<n;i++){
		cin>>a;
		cost.push_back(a);
	}
	v[0].push_back(1);
	for(int i=1;i<n;i++){
		cin>>a>>b;
		v[a].push_back(b);
		v[b].push_back(a);
	}
	cout<<dp(0,0);
	return 0;
}
```
