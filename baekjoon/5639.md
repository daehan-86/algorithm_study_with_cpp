## 풀이
이 문제는 전위순회의 결과를 보고 트리를 구성한 후  
후위 순회한 결과를 출력하면 되는 문제이다.
## 코드
```
#include<iostream>
using namespace std;
int a;
struct node {
	int n=0;
	node* r=NULL, *l=NULL;
	void push(int k) {
		if (!n) {
			n = k;
			return;
		}
		if (n > k) {
			if (l == NULL) {
				l = new node;
			}
			l->push(k);
		}
		else {
			if (r == NULL) {
				r = new node;
			}
			r->push(k);
		}
	}
	void print() {
		if (n) {
			if(l!=NULL)l->print();
			if (r != NULL)r->print();
			cout << n << "\n";
		}
	}
};

int main(void) {
	cin.tie(0);
	ios::sync_with_stdio(0);
	node t;
	while (cin >> a) {
		t.push(a);
	}
	t.print();
	return 0;
}
```
